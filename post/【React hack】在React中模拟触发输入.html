在 React 中进行模拟触发输入

参考：facebook/react#11488

在自动化测试或在线调试React项目中，会遇到直接修改input的valut值UI变化，但是内部state并没有修改的情况。因此需要hack方法来解决：

1// 修改input
2let input = document.querySelector('xxx');
3let lastValue = input.value;
4input.value = 'xxx';
5let event = new Event('input', {bubbles: true});
6event.simulated = true;
7let tracker = input._valueTracker;
8if (tracker) {
9  tracker.setValue(lastValue)
10}
11input.dispatchEvent(event);
12const enterKeyEvent = new KeyboardEvent('keypress', {
13  key: 'Enter',
14  keyCode: 13,
15  code: 'Enter',
16  which: 13,
17  bubbles: true,
18});
19input.dispatchEvent(enterKeyEvent);
1// 文件上传
2const fileInput = document.querySelector('input[type="file"]');
3
4async function simulateRemoteFileSelection(url) {
5  const response = await fetch(url);
6  const blob = await response.blob();
7
8  const file = new File([blob], "file.mp4", {
9    type: blob.type,
10  });
11
12  const dataTransfer = new DataTransfer();
13  dataTransfer.items.add(file);
14
15  fileInput.files = dataTransfer.files;
16
17  const event = new Event('change', {bubbles: true});
18  fileInput.dispatchEvent(event);
19}
20
21simulateRemoteFileSelection('xxxx');
